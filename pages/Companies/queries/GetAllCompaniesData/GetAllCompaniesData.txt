query CompanyDetails($limit: Int, $offset: Int, $searchText: String, $statusId: Int) {
	prod {
		missionctrl_map_company_to_status(
			limit: $limit
			offset: $offset
			where: {
				_and: [
					{ Company: { LegalNameOfBusiness: { _like: $searchText } } }
					{ Company: { TenantId: { _eq: "BB2090DC-81C1-49ED-AE2E-5016C79464AB" } } }
				]
			}
		) {
			Company {
				Id
				LegalNameOfBusiness
				Marketplaces {
					MarketplaceId
					MarketplaceTypeId
				}
				Country {
					Id
					NameEN
				}
				EstablishedCountry {
					Code
					NameEN
				}
				LegalStatus {
					NameEN
				}
				current_orders {
					status {
						name
					}
					subscription_status {
						status {
							name
							description
						}
					}
					country {
						NameEN
						Code
					}
					LastEndDate
				}
			}
			CompanyStatus {
				id
				name
			}
		}
		missionctrl_map_company_to_status_aggregate(
			where: {
				Company: { TenantId: { _eq: "BB2090DC-81C1-49ED-AE2E-5016C79464AB" } }
				_and: [{ status_id: { _eq: $statusId } }]
			}
		) {
			aggregate {
				count
			}
		}
	}
}
